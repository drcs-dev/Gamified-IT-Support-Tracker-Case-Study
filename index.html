Abstract

The Gamified IT Support Tracker is a custom ServiceNow application designed to enhance IT support operations through gamification-based incident tracking. The system awards points to support agents according to incident priority and SLA compliance.


The application uses Business Rules, GlideRecord scripting, scratchpad data loading, and dashboards to automate rewards and display badge-based performance metrics.

Badge Levels

  • Bronze — 0 to 99 Points
  • Silver — 100 to 199 Points
  • Gold — 200 to 299 Points
  • Platinum — 300+ Points


Priority-Based Points

  • Priority 1 — 50 Points
  • Priority 2 — 35 Points
  • Priority 3 — 20 Points
  • Priority 4 — 10 Points
  • Priority 5 — 5 Points

Business Rule — Award Points on Incident Closure

(function executeRule(current, previous) {

  if (!(current.state.changesTo('7') && !current.u_points_awarded))
            return;

            if (!current.assigned_to)
            return;

            var pointsMap = {
              '1': 50,
            '2': 35,
            '3': 20,
            '4': 10,
            '5': 5
  };

            var points = pointsMap[String(current.priority)] || 0;

            var slaBonus = 0;

            var ts = new GlideRecord('task_sla');
            ts.addQuery('task', current.sys_id);
            ts.addQuery('stage', 'completed');
            ts.addQuery('has_breached', false);
            ts.query();

            if (ts.next())
            slaBonus = 10;

            var total = points + slaBonus;

            var gr = new GlideRecord('u_agent_rewards');
            gr.addQuery('u_agent', current.assigned_to);
            gr.query();

            if (gr.next()) {
    var newTotal = (parseInt(gr.u_points, 10) || 0) + total;

            gr.u_points = newTotal;
            gr.u_badge = getBadge(newTotal);
            gr.update();
  } else {
              gr.initialize();
            gr.u_agent = current.assigned_to;
            gr.u_points = total;
            gr.u_badge = getBadge(total);
            gr.insert();
  }

            current.u_points_awarded = true;
            current.update();

            function getBadge(score) {
    if (score >= 300)
            return 'Platinum';

    if (score >= 200)
            return 'Gold';

    if (score >= 100)
            return 'Silver';

            return 'Bronze';
  }
})(current, previous);

Business Rule — Load Rewards to Scratchpad

(function executeRule(current, g_scratchpad) {

            g_scratchpad.badge = '';
            g_scratchpad.points = 0;

            if (!current.assigned_to)
            return;

            var r = new GlideRecord('u_agent_rewards');
            r.addQuery('agent', current.assigned_to);
            r.setLimit(1);
            r.query();

            if (r.next()) {
              g_scratchpad.badge = r.badge + '';
            g_scratchpad.points = parseInt(r.points, 10) || 0;
  }
})(current, g_scratchpad);

Client Script — Display Badge Information

function onLoad() {

  if (g_scratchpad.badge) {

              g_form.addInfoMessage(
                'Agent Badge: ' + g_scratchpad.badge +
                ' | Points: ' + g_scratchpad.points
              );
  }
}
